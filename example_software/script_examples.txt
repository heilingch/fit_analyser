def parse_fit(data):
    offset = 14  # skip header
    end = 14 + struct.unpack('<I', data[4:8])[0]

    definitions = {}
    records = []
    sessions = []

    FIT_EPOCH = 631065600  # seconds from Unix epoch to FIT epoch (Dec 31, 1989)

    while offset < end - 2:
        header_byte = data[offset]
        offset += 1

        is_definition = (header_byte & 0x40) != 0
        local_msg_type = header_byte & 0x0F

        if header_byte & 0x80:  # compressed timestamp
            continue

        if is_definition:
            reserved = data[offset]; offset += 1
            arch = data[offset]; offset += 1
            global_mesg_num = struct.unpack('<H' if arch == 0 else '>H', data[offset:offset+2])[0]
            offset += 2
            num_fields = data[offset]; offset += 1

            fields = []
            for _ in range(num_fields):
                field_def_num = data[offset]
                field_size = data[offset+1]
                base_type = data[offset+2]
                offset += 3
                fields.append((field_def_num, field_size, base_type))

            # Handle developer fields if present (header bit 5)
            if header_byte & 0x20:
                num_dev_fields = data[offset]; offset += 1
                for _ in range(num_dev_fields):
                    offset += 3  # skip dev field defs

            definitions[local_msg_type] = (global_mesg_num, fields, arch == 1)

        else:  # data message
            if local_msg_type not in definitions:
                break

            global_mesg_num, fields, big_endian = definitions[local_msg_type]

            record = {'_mesg': global_mesg_num}

            for field_def_num, field_size, base_type in fields:
                raw = data[offset:offset+field_size]
                offset += field_size

                # Determine field lookup
                if global_mesg_num == 20:  # record
                    field_info = RECORD_FIELDS.get(field_def_num)
                elif global_mesg_num == 18:  # session
                    field_info = SESSION_FIELDS.get(field_def_num)
                else:
                    field_info = None

                if field_info is None:
                    continue

                fname = field_info[0]
                scale = field_info[2] if len(field_info) > 2 else 1
                offset_val = field_info[3] if len(field_info) > 3 else 0

                # Get base type info
                bt = BASE_TYPES.get(base_type, None)
                if bt is None:
                    # try with just lower nibble
                    continue

                bname, bsize, bfmt = bt

                if bfmt and bsize == field_size:
                    endian = '>' if big_endian else '<'
                    fmt = bfmt.replace('<', endian).replace('>', endian)
                    try:
                        val = struct.unpack(fmt, raw)[0]

                        # Check for invalid values
                        invalid = {1: 0xFF, 2: 0xFF, 4: 0xFFFFFFFF, 8: 0xFFFFFFFFFFFFFFFF}
                        inv_signed = {1: 0x7F, 2: 0x7FFF, 4: 0x7FFFFFFF, 8: 0x7FFFFFFFFFFFFFFF}

                        if val == invalid.get(bsize, None) or val == inv_signed.get(bsize, None):
                            continue

                        if fname == 'timestamp':
                            record[fname] = val + FIT_EPOCH  # convert to unix
                        elif fname in ('position_lat', 'position_long'):
                            record[fname] = val * (180.0 / 2**31)
                        elif fname == 'altitude':
                            record[fname] = val * 0.2 - 500
                        elif fname == 'avg_altitude':
                            record[fname] = val * 0.2 - 500
                        elif fname == 'max_altitude':
                            record[fname] = val * 0.2 - 500
                        elif fname == 'min_altitude':
                            record[fname] = val * 0.2 - 500
                        else:
                            record[fname] = val * scale + offset_val
                    except:
                        continue

            if global_mesg_num == 20 and len(record) > 2:
                records.append(record)
            elif global_mesg_num == 18 and len(record) > 2:
                sessions.append(record)

    return records, sessions
